  **1. Add token refresh to the existing scheduler.**

  Find the existing `setInterval` that processes due posts (search for `processDuePosts` or `setInterval`). Add a token refresh check that runs alongside it. The simplest approach: add a counter that triggers token refresh every 60 iterations (once per hour if scheduler runs every minute):

  ```typescript
  import { refreshInstagramToken } from "./instagram";

  let tokenRefreshCounter = 0;

  // Inside the existing setInterval callback, AFTER processDuePosts:
  tokenRefreshCounter++;
  if (tokenRefreshCounter >= 60) {
    tokenRefreshCounter = 0;
    try {
      const allCredentials = await storage.getAllInstagramCredentials();
      for (const cred of allCredentials) {
        if (!cred.tokenExpiresAt) continue;
        const msUntilExpiry = cred.tokenExpiresAt.getTime() - Date.now();
        const tenDaysMs = 10 * 24 * 60 * 60 * 1000;
        if (msUntilExpiry > tenDaysMs) continue;
        if (msUntilExpiry <= 0) {
          await storage.markInstagramTokenExpired(cred.userId);
          continue;
        }
        const result = await refreshInstagramToken(cred.accessToken);
        if (result) {
          const newExpiresAt = new Date(Date.now() + result.expiresIn * 1000);
          await storage.updateInstagramToken(cred.userId, result.accessToken, newExpiresAt);
          console.log(`[Instagram] Refreshed token for user ${cred.userId}`);
        } else {
          await storage.markInstagramTokenExpired(cred.userId);
          console.warn(`[Instagram] Token refresh failed for user ${cred.userId} — marked expired`);
        }
      }
    } catch (err) {
      console.error("[Instagram] Token refresh error:", err);
    }
  }
  ```
